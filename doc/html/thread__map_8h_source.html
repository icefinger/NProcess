486  operator[](const key_type& __k)
487  {
488  // concept requirements
489  pthread_mutex_lock(&_main_mutex2);
490  __glibcxx_function_requires(_DefaultConstructibleConcept<mapped_type>)
491 
492  iterator __i = lower_bound(__k);
493 
494  // __i->first is greater than or equivalent to __k.
495  if (__i == end() || key_comp()(__k, (*__i).first))
496  __i = insert(__i, value_type(__k, mapped_type()));
497  mapped_type& to_return=(*__i).second;
498  pthread_mutex_unlock(&_main_mutex2);
499  return to_return;
500  }
